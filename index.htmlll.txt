    <script>
        // --- Cursor Movement ---
        const cursor = document.getElementById('cursor');
        document.addEventListener('mousemove', (e) => {
            cursor.style.left = e.clientX + 'px';
            cursor.style.top = e.clientY + 'px';
        });

        // --- GSAP Scroll Registration ---
        gsap.registerPlugin(ScrollTrigger);

        function startJourney() {
            // Play Audio
            const music = document.getElementById('bgMusic');
            music.volume = 0.3;
            music.play();

            // Transition Landing
            gsap.to("#starter", { duration: 1, opacity: 0, y: -50, ease: "power4.in" });
            gsap.to("#landing", { duration: 1.5, delay: 0.5, opacity: 0, display: "none" });

            // Reveal Sections
            document.querySelectorAll('.hidden').forEach(el => {
                el.classList.remove('hidden');
            });

            // Journey Animations
            gsap.from(".bento-item", {
                scrollTrigger: {
                    trigger: "#journey",
                    start: "top center",
                },
                duration: 1,
                y: 100,
                opacity: 0,
                stagger: 0.2,
                ease: "power4.out"
            });

            // Scroll to next section automatically
            setTimeout(() => {
                window.scrollTo({ top: window.innerHeight, behavior: 'smooth' });
            }, 1000);
        }

        function makeWish() {
            // Blow out flame
            gsap.to("#birthdayFlame", { 
                duration: 0.5, 
                scale: 0, 
                opacity: 0, 
                onComplete: () => {
                    document.getElementById('birthdayFlame').style.display = 'none';
                    document.getElementById('wishInstruction').innerText = "Wish Granted ✨";
                    
                    // Trigger Confetti
                    confetti({
                        particleCount: 150,
                        spread: 70,
                        origin: { y: 0.6 },
                        colors: ['#ff3366', '#7000ff', '#ffffff']
                    });

                    // Reveal Final Message
                    setTimeout(() => {
                        window.scrollTo({ 
                            top: document.body.scrollHeight, 
                            behavior: 'smooth' 
                        });
                    }, 1500);
                }
            });
        }

        // Add subtle parallax to items
        document.addEventListener('mousemove', (e) => {
            const x = (window.innerWidth / 2 - e.pageX) / 50;
            const y = (window.innerHeight / 2 - e.pageY) / 50;
            gsap.to(".bento-item", { x: x, y: y, duration: 1 });
        });
    </script>
